def fibonacci(limit):
    fib = [0, 1]
    while fib[-1] < limit:
        next_value = fib[-1] + fib[-2]
        fib.append(next_value)
    return fib
    
def check_fibonacci(number):
    fib = fibonacci(number)
    if number in fib:
        return f"O numero {number} pertence a sequência de Fibonacci!"
    else:
            return f"O número {number} não pertence a sequência de Fibonacci."
            

numero = int(input("Digite um numero: "))
resultado = check_fibonacci(numero)
print(resultado)
